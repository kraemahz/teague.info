\section{Notation Summary}
\label{app:notation}

Table~\ref{tab:notation7} collects the principal symbols used in
the paper.  Symbols inherited from earlier
papers~\citep{lasser2026poset, lasser2026horizon,
lasser2026scm, lasser2026exo, lasser2026phase} retain their
meanings and are cited rather than redefined.

\begin{table}[h]
\centering
\small
\begin{tabular}{@{}p{3.0cm}p{7.5cm}p{4.2cm}@{}}
\toprule
\textbf{Symbol} & \textbf{Definition} & \textbf{Introduced in} \\
\midrule
\multicolumn{3}{l}{\emph{Inherited machinery}} \\
$\volL$
  & Capability volume
  & \citep[Axiom system]{lasser2026poset} \\
$G_t$
  & Global capability graph at time $t$
  & \citep{lasser2026horizon} \\
$P$, $P_t$
  & Capability poset (full / at time $t$)
  & \citep{lasser2026poset, lasser2026phase} \\
$\Pathway$
  & Exercise pathway
  & \citep[Def.~2]{lasser2026horizon} \\
$\Risk$, $\Risk_j$
  & Concentration risk (total / keystone-indexed)
  & \citep[Def.~3]{lasser2026horizon} \\
$a_j$
  & Agent index
  & \citep{lasser2026horizon} \\
$T^{\mathrm{risk}}_j$
  & Risk-trust factor of agent $a_j$
  & \citep[Def.~4]{lasser2026scm} \\
\midrule
\multicolumn{3}{l}{\emph{Risk-restriction pair and observation regimes}} \\
$R$
  & Risk claim
    $(a_j, (d_1, \ldots, d_k), \Pathway, \pR)$
  & Def.~\ref{def:risk_restriction} \\
$\Xirestrict$
  & Preemptive restriction of a keystone capability
  & Def.~\ref{def:risk_restriction} \\
$\pR$, $\pR(t)$
  & Posterior probability the risk claim is valid
  & Def.~\ref{def:risk_restriction} \\
Restricted regime
  & $\Xirestrict$ active; pathway cannot activate
  & Def.~\ref{def:obs_regime} \\
Exercise regime
  & $\Xirestrict$ lifted; pathway may activate
  & Def.~\ref{def:obs_regime} \\
$p_{\mathrm{thresh}}$
  & Restriction-triggering threshold for $\pR$
  & Prop.~\ref{prop:monotone_overcaution} \\
\midrule
\multicolumn{3}{l}{\emph{Controlled relaxation test object}} \\
$T$
  & Controlled relaxation test
    $(R, \Xirestrict, \scope, \testdur, \damagebound)$
  & Def.~\ref{def:cr_test} \\
$\scope \subseteq P$
  & Test scope (bounded sub-poset)
  & Def.~\ref{def:cr_test}, Def.~\ref{def:test_scope} \\
$\testdur \in \mathbb{N}$
  & Test duration (steps)
  & Def.~\ref{def:cr_test} \\
$\damagebound > 0$
  & Damage tolerance (max.\ acceptable $\volL$-contraction)
  & Def.~\ref{def:cr_test} \\
$\maxcon(\scope)$
  & Max.\ $\volL$-contraction over all pathways in $\scope$
  & Def.~\ref{def:test_scope}, Lem.~\ref{lem:computable_maxcon} \\
$\deltairrev(\Pathway)$
  & Irreversible fraction of a pathway's contraction
  & Def.~\ref{def:test_scope}(iii) \\
$\deltairrev^{\max}$
  & $\max_{\Pathway} \deltairrev(\Pathway)$ within scope
  & Thm.~\ref{thm:damage_bound}(b) \\
$\partial \scope$
  & Cooperative surface of $\scope$
  & App.~\ref{app:proof_damage_bound} \\
\midrule
\multicolumn{3}{l}{\emph{Binary observation model}} \\
$A_n \in \{0,1\}$
  & Latent pathway-activation indicator, test $n$
  & Sec.~\ref{sec:relaxation_protocol} \\
$O_n \in \{0,1\}$
  & Observed contraction indicator, test $n$
  & Sec.~\ref{sec:relaxation_protocol} \\
$H_0$
  & Pathway-harmless hypothesis ($\theta = 0$)
  & Sec.~\ref{sec:relaxation_protocol} \\
$H_1$
  & Pathway-dangerous hypothesis ($\theta = \thetaone$)
  & Sec.~\ref{sec:relaxation_protocol} \\
$\thetaone \in (0,1]$
  & Fixed activation rate under $H_1$
  & Sec.~\ref{sec:relaxation_protocol} \\
$\beta$, $\beta(\testdur)$
  & False-negative rate over test window

    Def.~\ref{def:detection_sensitivity} \\
$\alphafp$
  & Per-test false-positive rate
  & Sec.~\ref{sec:relaxation_protocol} \\
$\Lzero$, $\Lone$
  & Per-test likelihood ratios for $O=0$ and $O=1$
  & Eq.~\eqref{eq:L_defs} \\
$\logit p = \log(p / (1-p))$
  & Log-odds transform
  & Eq.~\eqref{eq:update_pos} \\
